## 🔧 工具配置

### 必需依赖

```json
{
  "devDependencies": {
    "unocss": "^66.0.0",
    "@uni-helper/unocss-preset-uni": "^0.2.11"
  }
}
```

### 推荐配置

```typescript
// uno.config.ts
import { presetUni } from '@uni-helper/unocss-preset-uni';
import { defineConfig } from 'unocss';

export default defineConfig({
  presets: [presetUni()],
  // 添加你的自定义配置...
});
```
